Outlier Detection using LOC (Local Outlier Factor) [10 Marks]¶

LOF uses density-based outlier detection to identify local outliers. An anomaly score is assigned to each sample which measures the local deviation of density of a this sample with respect to its neighbors.

In [132]:
# Code here

def CustomLocalOutlierFactor(X, k):

    # Compute the k-distance for each data point
    nbrs = NearestNeighbors(n_neighbors=k+1).fit(X)
    distances, indices = nbrs.kneighbors(X)
    k_distances = distances[:, -1]

    # Compute the reachability distance for each pair of data points
    reachability_distances = np.zeros((len(X), k))
    for i in range(len(X)):
        for j in range(k):
            neighbor_index = indices[i, j+1]
            reachability_distances[i, j] = max(k_distances[neighbor_index], distances[i, j+1])
    
    # Compute the local reachability density for each data point
    lrd = k / np.sum(reachability_distances, axis=1)

    # Compute the LOF for each data point
    lof = np.zeros(len(X))
    for i in range(len(X)):
        lrd_ratios = lrd[indices[i, 1:]] / lrd[i]
        lof[i] = np.mean(lrd_ratios)

    return lof
