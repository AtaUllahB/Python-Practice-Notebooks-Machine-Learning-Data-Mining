def Loss_func(X, actual, reg_para, theta):
  z = hypothesis(X,theta)  
  loss = - ( (actual.T.dot(np.clip(np.log(sigmoid_mapping(z)), a_min = 0.0000000000001, a_max = 1- 0.0000000000001))) + ( (1-actual).T.dot(np.clip(np.log(1-sigmoid_mapping(z)), a_min = 0.0000000000001, a_max = 1- 0.0000000000001)) ) )
  return loss + reg_para * norm(theta)

# Implement Evaluation Function here
def Evaluation(labels, predicted): 
    # Calculate confusion matrix
    confusion = np.zeros((2, 2))
    for i in range(len(labels)):
        confusion[labels[i], predicted[i]] += 1
    
    # Calculate classification accuracy
    accuracy = np.sum(np.diag(confusion)) / np.sum(confusion)
    
    # Calculate precision, recall, and F1 score for each class
    precision = np.zeros(2)
    recall = np.zeros(2)
    f1_score = np.zeros(2)
    for i in range(2):
        tp = confusion[i, i]
        fp = np.sum(confusion[:, i]) - tp
        fn = np.sum(confusion[i, :]) - tp
        precision[i] = tp / (tp + fp)
        recall[i] = tp / (tp + fn)
        f1_score[i] = 2 * precision[i] * recall[i] / (precision[i] + recall[i])
    
    return confusion, accuracy, f1_score
